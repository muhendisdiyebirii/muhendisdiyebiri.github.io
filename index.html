Sayı Tahmin Oyunu

Bilgisayarın 1-100 arasında tuttuğu sayıyı tahmin etmeye çalıştığınız bir oyun.

import random

def tahmin():
  sayi = random.randint(1, 100)
  while True:
    a = int(input("Bir sayı seçiniz:"))
    if sayi == a:
      print("Tahmininiz doğru!")
      break
    if sayi >= a:
      print("Daha Büyük bir sayı giriniz:")
    if sayi <= a:
      print("Daha Küçük bir sayı giriniz:")

tahmin()

Taş-Kağıt-Makas Oyunu

Bilgisayara karşı oynayabileceğiniz basit bir Taş-Kağıt-Makas oyunu.

import random

def secim():
    secenekler = ["Taş", "Kağıt", "Makas"]
    bilgisayar_secimi = random.choice(secenekler)
    oyuncu_secimi = input("Hadi bakalım! Taş mı, Kağıt mı, yoksa Makas mı? Seçimini yap: ").capitalize()

    if oyuncu_secimi not in secenekler:
        print("Öyle bir seçenek yok! Lütfen sadece 'Taş', 'Kağıt' veya 'Makas' yaz.")
        return

    print(f"Bilgisayarın tercihi: {bilgisayar_secimi}")
    print(f"Senin tercihin: {oyuncu_secimi}")

    if bilgisayar_secimi == oyuncu_secimi:
        print("Berabere! Bu seferlik kimse kazanmadı.")
    elif (bilgisayar_secimi == "Taş" and oyuncu_secimi == "Makas") or \
         (bilgisayar_secimi == "Kağıt" and oyuncu_secimi == "Taş") or \
         (bilgisayar_secimi == "Makas" and oyuncu_secimi == "Kağıt"):
        print("Kaybettin! Daha iyisini yapabilirsin.")
    else:
        print("Tebrikler! Kazandın. Helal olsun!")

while True:
    secim()
    devam = input("Devam etmek istiyor musun? (E/h): ").lower()
    if devam != "e":
        print("Oyun bitti! Yine bekleriz.")
        break

Rastgele Şifre Oluşturucu

Belirttiğiniz uzunlukta güçlü ve rastgele şifreler oluşturan bir script.

import random
import string

# Kullanıcıdan şifre uzunluğunu alma
uzunluk = int(input("Şifrenizin Olmasını İstediğiniz Uzunluğu Giriniz:"))

def sifre():
    # Şifre elemanlarını belirle: büyük/küçük harf, rakam ve özel karakterler
    eleman = string.ascii_letters + string.digits + "!#+%&_"
    # Random olarak şifre oluştur
    şifre = ''.join(random.choice(eleman) for i in range(uzunluk))
    # Oluşturulan şifreyi ekrana yazdır
    print("Şifreniz: " + şifre)

# Fonksiyonu çağır
sifre()

Şifreli Liste Yöneticisi

Kullanıcının belirlediği şifre ile listeye eleman ekleme, çıkarma ve görüntüleme işlemleri yapılabilen basit bir yönetim sistemi.

sifre = int(input("Bir Şifre Belirleyiniz:"))

def islem():
    print("İşlem Başarılı!")

def sifre_kontrol():
    giris = int(input("Şifreyi Giriniz: "))
    if giris == sifre:
        return True
    else:
        print("Hatalı Şifre!")
        return False

liste = []  

while True:
    secim = input("1.Listeye Ekle\n2.Listeden Çıkar\n3.Listeyi Göster\n4.Kapat\nSeçiminizi Yapınız:")

    if secim == "1":
        if sifre_kontrol():  
            ekle = input("Eklemek İstediğin Aktiviteyi Giriniz: ")
            liste.append(ekle)
            print(ekle + " eklendi")
            islem()

    elif secim == "2":
        if sifre_kontrol():
            print(liste)
            cikar = input("Listeden Çıkartmak İstediğiniz Aktiviteyi Giriniz: ")
            if cikar in liste:
                liste.remove(cikar)
                islem()
            else:
                print("Bu aktivite listede yok!")

    elif secim == "3":
        if sifre_kontrol():
            print(liste)

    elif secim == "4":
        print("Program Kapatılıyor...")
        break

Hesap Makinesi

Toplama, çıkarma, çarpma, bölme, üs alma, faktöriyel ve karekök işlemlerini yapabilen gelişmiş bir hesap makinesi.

import math

def sayi():
    return float(input("İlk Sayıyı Giriniz:"))

def sayi2():
    return float(input("İkinci Sayıyı Giriniz:"))

while True:

    secim = input("1.Toplama İşlemi\n2.Çıkarma İşlemi\n3.Çarpma İşlemi\n4.Bölme İşlemi\n5.Üs Alma\n6.Faktöriyel Alma\n7.Karekök Alma\n8.Kapat\nLütfen Yapmak İstediğiniz İşlemin Numarasını Giriniz:")

    if secim == "1":
        t1 = sayi()
        t2 = sayi2()
        print(t1 + t2)
    if secim == "2":
        ç1 = sayi()
        ç2 = sayi2()
        print(ç1 - ç2)
    if secim == "3":
        ça1 = sayi()
        ça2 = sayi2()
        print(ça1 * ça2)
    if secim == "4":
        b1 = sayi()
        b2 = sayi2()
        print(b1 / b2)
    if secim == "5":
        ü1 = sayi()
        ü2 = sayi2()
        print(ü1 ** ü2)
    if secim == "6":
        f = int(input("Faktöriyel almak istediğiniz sayıyı giriniz:"))
        f = math.factorial(f)
        print(f)
    if secim == "7":
        k = float(input("Karekökünü Almak İstediğiniz Sayıyı Giriniz:"))
        k = math.sqrt(k)
        print(k)
    if secim == "8":
        print("Program Kapatılıyor...")
        break
